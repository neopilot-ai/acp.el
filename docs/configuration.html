Configuration

Customize acp.el to fit your workflow.

Basic Setup

(require 'acp)

;; Set preferred agent (shown on M-x acp)
(setq acp-preferred-agent-config
      (acp-anthropic-make-claude-code-config))

Agent Authentication

Anthropic (Claude Code)

;; API key authentication
(setq acp-anthropic-authentication
      (acp-anthropic-make-authentication :api-key "sk-ant-..."))

;; Browser-based login (recommended)
(setq acp-anthropic-authentication
      (acp-anthropic-make-authentication :login t))

Google (Gemini CLI)

;; Uses gcloud ADC — run once:
;; gcloud auth application-default login
(setq acp-google-authentication
      (acp-google-make-authentication :login t))

OpenAI (Codex)

;; API key
(setq acp-openai-authentication
      (acp-openai-make-authentication :api-key "sk-..."))

Environment Variables

Pass custom environment variables to agents:

(setq acp-anthropic-claude-environment
      (acp-make-environment-variables
       "HTTPS_PROXY" "http://proxy.example.com:8080"
       "HTTP_PROXY"  "http://proxy.example.com:8080"))

;; Inherit the parent Emacs process environment
(setq acp-anthropic-claude-environment
      (acp-make-environment-variables :inherit-env t))

;; Load from .env file
(setq acp-anthropic-claude-environment
      (acp-make-environment-variables :load-env "~/.env"))

MCP Servers

Configure Model Context Protocol servers for extended tool access:

(setq acp-mcp-servers
  '(; Filesystem access — restrict to home directory
    ((name . "filesystem")
     (type . file-system)
     (config . ((allowedDirectories . ["~/"]))))

    ; HTTP-based MCP server
    ((name . "notion")
     (type . http)
     (url . "https://mcp.notion.com/mcp"))

    ; stdio-based server
    ((name . "custom-tool")
     (type . stdio)
     (command . "my-mcp-server")
     (args . ("--port" "3000")))))

Session Strategy

Control how sessions are created and resumed:

;; 'latest — resume the most recent session (default)
;; 'new    — always create a new session
;; 'list   — prompt to select from available sessions
(setq acp-session-strategy 'latest)

Transcripts

Sessions are automatically saved as Markdown transcripts:

;; Default: /.acp/transcripts/YYYY-MM-DD-HH-MM-SS.md
;; Disable transcripts:
(setq acp-transcript-file-path-function nil)

;; Custom path function:
(setq acp-transcript-file-path-function
      (lambda () (expand-file-name
                  (format-time-string "session-%s.md")
                  "~/acp-transcripts/")))

Container Support

Run agents inside Docker containers or devcontainers:

;; Docker image
(setq acp-docker-enabled t)
(setq acp-docker-image "ubuntu:latest")
(setq acp-docker-workdir "/workspace")

;; Docker Compose
(setq acp-docker-compose-file "docker-compose.yml")
(setq acp-docker-service "agent")

;; Devcontainer
(setq acp-command-prefix '("devcontainer" "exec" "--workspace-folder" "."))
(setq acp-path-resolver-function #'acp-devcontainer-resolve-path)

UI Customization

Context Usage Indicator

;; Show context window usage percentage in modeline
(setq acp-show-context-usage-indicator 'detailed)  ;; "29k/200k (15%)"
(setq acp-show-context-usage-indicator t)           ;; Bar: █▇▆▅▄▃▂▁
(setq acp-show-context-usage-indicator nil)          ;; Hidden

Usage at Turn End

;; Display token usage summary after each agent response
(setq acp-show-usage-at-turn-end t)

File Tree Browser

;; Side panel location
(setq acp-file-tree-side 'left)   ;; or 'right

;; Panel width
(setq acp-file-tree-width 30)

;; Ignored directories
(setq acp-file-tree-ignored-directories
      '(".git" "node_modules" "__pycache__" ".acp"))

Permission Manager

;; Default policy for unspecified actions
(setq acp-permissions-default-policy 'prompt)  ;; or 'allow, 'deny

;; File read policy (default: allow)
(setq acp-permissions-file-read-policy 'allow)

;; File write policy (default: prompt)
(setq acp-permissions-file-write-policy 'prompt)

;; Command execution policy (default: prompt)
(setq acp-permissions-command-execution-policy 'prompt)

;; Always-allowed paths
(setq acp-permissions-allowed-paths '("~/projects/"))

;; Always-denied paths (takes precedence)
(setq acp-permissions-denied-paths '("/etc/" "/root/"))

;; Enable audit logging
(setq acp-permissions-audit-log t)
